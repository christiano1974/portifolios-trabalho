  #### Este programa (função chamada de numero_palindromo) escrita na linguagem R, verifica 
  ###  se um número é palindromo tendo como parâmetros o número e a quantidade de
  ###  algarismos que o número contém. Um número é palíndromo quando ao invertemos os seus   ###  algarismo o número não se altera.
  ###  Por exemplo: 11, 434, 2772, assim por diante ...

  ### programa escrito em 05/11  ,  por Christiano Garcia.


   numero_palindromo = function(num, n_algarismo){

            num = as.character(num)
            vetor_num = vector("character",n_algarismos)
            for(k in 1:nchar(num))vetor_num[k]=substr(num,k,k)
            inverso = vector("character",n_algarismos)
            k = nchar(num)
            while(k > 0){
              inverso[k] = vetor_num[1 + nchar(num) - k]
              k = k - 1}     
            cont = 0
            for(k in 1:nchar(num)){
                if ( vetor_num[k] == inverso[k]) cont = cont + 1 }
            if ( cont == n_algarismos){
                print('Número digitado é palíndromo !!!')}
            else{
                print('Número digitado não é palíndromo !!!')}          
            
    }
